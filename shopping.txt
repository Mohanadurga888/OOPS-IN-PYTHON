class Customer:
    # Initialize the customer with an empty list of items
    def __init__(self):
        self.items = []

    # Add an item with a name and quantity to the customer
    def add_item(self, item_name, qty):
        item = (item_name, qty)
        self.items.append(item)

    # Remove an item with a specific name from the shopping cart
    def remove_item(self, item_name):
        for item in self.items:
            if item[0] == item_name:
                self.items.remove(item)
                break

    # Calculate the total quantity of items
    def calculate_total(self):
        total = 0
        for item in self.items:
            total += item[1]
        return total

# Create an instance of Customer
d = Customer()

# Add items to the shopping cart
d.add_item("Papaya", 100)
d.add_item("Guava", 200)
d.add_item("Orange", 150)

# Display the current items in the cart and calculate the total quantity
print("Current Items in Cart:")
for item in d.items:
    print(item[0], "=", item[1])

total_qty = d.calculate_total()
print("Total Quantity:", total_qty)

# Remove an item and show updated cart
d.remove_item("Guava")
print("\nUpdated Items in Cart after removing Guava:")
for item in d.items:
    print(item[0], "-", item[1])

total_qty = d.calculate_total()
print("Total Quantity:", total_qty)
